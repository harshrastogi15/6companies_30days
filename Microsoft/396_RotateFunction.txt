class Solution {
public:
    int maxRotateFunction(vector<int>& nums) {
        int totalsum=0;
        int sum=0;
        for(int i=0;i<nums.size();i++){
            sum += nums[i];
            totalsum += i*nums[i];
        }
        int mx = totalsum;
        int n= nums.size()-1;
        for(int i=nums.size()-1;i>=0;i--){
            totalsum -= nums[i]*n;
            totalsum += sum - nums[i];
            mx = max(totalsum,mx);
        }
        return mx;
    }
};
